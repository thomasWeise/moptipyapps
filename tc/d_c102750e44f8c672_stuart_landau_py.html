36def __stuart_landau_equations(state: np.ndarray, _: float, 

37 control: np.ndarray, out: np.ndarray) -> None: 

38 """ 

39 Compute the differential equations of a controlled 2D Stuart-Landau model. 

40 

41 :param state: the state of the system 

42 :param _: the time index, which is ignored 

43 :param control: the output of the controller 

44 :param out: the differential, i.e., the output of this function 

45 """ 

46 sigma: Final[float] = 0.1 - state[0]**2 - state[1]**2 

47 out[0] = sigma * state[0] - state[1] 

48 out[1] = sigma * state[1] + state[0] + control[0]
